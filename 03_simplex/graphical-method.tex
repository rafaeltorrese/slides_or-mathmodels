\begin{frameExample}{Graphical Method Of Solution}{}
  \[ \max Z = 3x_1 + 4x_2\]

  {\centering
    subject to (s.t.)

    \sysalign{r,r}%
    \sysdelim..%
    \systeme[x_1x_2]%
    {
      x_1 + x_2 \leq 450,
      2x_1 + x_2 \leq 600
    }

    \vspace{5mm}
    $x_1, x_2 \geq 0$
  \par}
\end{frameExample}
