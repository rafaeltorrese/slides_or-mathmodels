\begin{frameExample}{Trial and Error Method \label{example:02.15-02}}{}
  % EXAMPLE  2.15-2 Gupta Ebook

\[     \max Z = x_1 + 3x_2 + 3x_3\]

{\centering
  subject to

  \sysdelim..%
  \sysalign{r,r}%
  \systeme[x_1x_2x_3]%
  {
    x_1 + 2x_2 + 3x_3 = 4,
    2x_1 + 3x_2 + 5x_3 = 7
  }

  \vspace{3mm}
  $x_1, x_2,x_3  \geq 0$
  \par}
\end{frameExample}

\begin{frameExample}{Trial and Error Method. \label{example:02.09-07}}{}
  \begin{columns}[t]
    \column{0.4\textwidth}
    % EXAMPLE  2.09-7 Gupta Ebook
    \[\min Z = -x_1 + 2x_2 \]

    {
      \centering
      subject to

      \sysdelim..%
      \sysalign{r,r}%
      \systeme[x_1x_2]%
      {
        -x_1 + 3x_2  \leq 10,
        x_1 + x_2  \leq 6,
        x_1 - x_2  \leq 2
      }

      \vspace{4mm}
      $x_1, x_2  \geq 0$
      \par
    }
    \column{0.6\textwidth}

    \[\min Z = -x_1 + 2x_2 + 0s_1 + 0s_2 + 0s_3 \]

    {\centering
      subject to
      
      \sysdelim..%
      \sysalign{r,r}%
      \systeme[x_1x_2s_1s_2s_3]%
      {
        -x_1 + 3x_2 + s_1  = 10,
        x_1 + x_2 + s_2 = 6,
        x_1 - x_2 + s_3 = 2
      }
      
   \vspace{4mm}
   $x_1, x_2, s_1, s_2, s_3  \geq 0$
\par}
  \end{columns}
\end{frameExample}
