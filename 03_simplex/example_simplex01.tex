  \begin{onlyenv}<1>
Solve the example~\ref{example:2-9-3_Gupta-ebook} with the simplex algorithm
  \begin{columns}
    \column{0.4\textwidth}
    \[ \max Z = 2x_1 + x_2 \]
      
    {\centering
      subject to

      \sysdelim..%
      \sysalign{r,r}%
      \systeme[x_1x_2]%
      {
        x_1 + 2x_2  \leq 10,
        x_1 + x_2  \leq 6,
        x_1 - x_2  \leq 2,
        x_1 - 2x_2  \leq 1
      }

      \vspace{5mm}
      $x_1, x_2  \geq 0$
      \par}
    \column{0.6\textwidth}
    \[ \max Z = 2x_1 + x_2 \]

    {\centering
      s.t.%
      
        \sysdelim..%
        \sysalign{r,r}%
        \systeme[x_1x_2s_1s_2s_3s_4]%
        {%
          x_1 + 2x_2 + s_1  = 10,
          x_1 + x_2 + s_2 = 6,
          x_1  - x_2  + s_3 = 2,
          x_1  - 2x_2  + s_4 = 1
        } % systeme

      \vspace{5mm}
      $x_1, x_2, s_1, s_2, s_3, s_4  \geq 0$
      \par}
  \end{columns}
  \end{onlyenv}
