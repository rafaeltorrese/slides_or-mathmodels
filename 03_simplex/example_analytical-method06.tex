\begin{frameExample}{Trial And Error Method  \label{example:2-9-3_Gupta-ebook}}{}

  % EXAMPLE  2.9-3 Gupta ebook
  \[     \max Z = 2x_1 + x_2 \]

  {\centering
    subject to

    \sysdelim..%
    \sysalign{r,r}%
    \systeme[x_1x_2]%
    {
      x_1 + 2x_2  \leq 10,
      x_1 + x_2  \leq 6,
      x_1 - x_2  \leq 2,
      x_1 - 2x_2  \leq 1
    }

    \vspace{5mm}
    $    x_1, x_2  \geq 0$
    \par}


\end{frameExample}
