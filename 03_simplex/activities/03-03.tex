    \question
    Resolver por método simplex el siguiente modelo.

    \[\max Z = 5x_1 + 4x_2\]

    {\centering
      subject to
    
      \sysdelim..\sysalign{r,r}\systeme[x_1x_2]%
      {
        6x_1 + 4x_2  \leq 24,
        x_1 + 2x_2  \leq 6,
        -x_1 + x_2  \leq 1,
        x_2  \leq 2
      }
    
      \vspace{5mm}
      $x_1, x_2  \geq 0$
      \par}

    \begin{solution}
        
        Initial basic feasible solution: 

        \[\max Z = 5x_1 + 4x_2  + 0s_1 + 0s_2 + 0s_3 + 0s_4\]

{\centering
  subject to

  \sysdelim..\sysalign{r,r}\systeme[x_1x_2s_1s_2s_3s_4]%
  {
    6x_1 + 4x_2 + s_1  = 24,
    x_1 + 2x_2  + s_2 = 6,
    -x_1 + x_2  + s_3 = 1,
    x_2  + s_4 = 2
  }

  \vspace{5mm}
  $x_1, x_2, s_1, s_2, s_3, s_4  \geq 0$
  \par}  


    \end{solution}

    \vspace{1cm}
    \question
    Resolver por método simplex el siguiente modelo. Usar método de Penalización.
    \[\max Z = 3x_1 + 4x_2 \]

    {\centering
      subject to (s.t.)
  
      \sysalign{r,r}%
      \sysdelim..%
      \systeme[x_1x_2]%
      {
        5x_1 + 4x_2 \leq 200,
        3x_1 + 5x_2 \leq 150,
        5x_1 + 4x_2 \geq 100,
        8x_1 + 4x_2 \geq 80
      }
  
      \vspace{5mm}
  
      $x_1 , x_2 \geq 0$
    \par}

   \begin{solution}
       
    Standard form: 

    \[\max Z = 3x_1 + 4x_2 + 0s_1 + 0s_2 + 0s_3 + 0s_4 \]

    {\centering
      subject to (s.t.)
  
      \sysalign{r,r}%
      \sysdelim..%
      \systeme[x_1x_2s_1s_2s_3s_4]%
      {
        5x_1 + 4x_2 + s_1 =   200,
        3x_1 + 5x_2 + s_2 = 150,
        5x_1 + 4x_2 - s_3 = 100,
        8x_1 + 4x_2 - s_4 = 80
      }
  
      \vspace{5mm}
  
      $x_1 , x_2, s_1, s_2, s_3, s_4 \geq 0$
    \par}
    

    Initial feasible basic solution with artificial variables:

    \[\max Z = 3x_1 + 4x_2 + 0s_1 + 0s_2 + 0s_3 + 0s_4 - MA_3 - MA_4\]

    {\centering
      subject to (s.t.)
  
      \sysalign{r,r}%
      \sysdelim..%
      \systeme[x_1x_2s_1s_2s_3s_4A_3A_4]%
      {
        5x_1 + 4x_2 + s_1 =   200,
        3x_1 + 5x_2 + s_2 = 150,
        5x_1 + 4x_2 - s_3 + A_3= 100,
        8x_1 + 4x_2 - s_4 + A_4= 80
      }
  
      \vspace{5mm}
  
      $x_1 , x_2, s_1, s_2, s_3, s_4, A_3, A_4 \geq 0$
    \par}
   \end{solution}

   \question
   Resolver el problema de la dieta con el método de las dos fases.

   \[ \min Z = 45x_1 + 40x_2 + 85x_3 + 65x_4   \]

    {\centering
      s.t.  %
      
    
      \sysdelim..\sysalign{r,r}%
      \systeme[x_1x_2x_3x_4]%
      {%
        3x_1 + 4x_2 + 8x_3 + 6x_4 \geq  800,
        2x_1 + 2x_2 + 7x_3 + 5x_4 \geq 200,
        6x_1  +4x_2 + 7x_3 + 4x_4 \geq 700%
      }

    $x_1, x_2, x_3, x_4 \geq 0$
  \par}

  \begin{solution}
    
    \[ \min W = 0x_1 + 0x_2 + 0x_3 + 0x_4 - 0s_1 - 0s_2 - 0s_3  + A_1 + A_2 + A_3 \]

    {\centering
      s.t.  %
      
      \sysdelim..\sysalign{r,r}%
      \systeme[x_1x_2x_3x_4s_1s_2s_3A_1A_2A_3]{%
        3x_1 + 4x_2 + 8x_3 + 6x_4 - s_1 + A_1 = 800,
        2x_1 + 2x_2 + 7x_3 + 5x_4 - s_2 + A_2 = 200,
        6x_1  +4x_2 + 7x_3 + 4x_4 - s_3 + A_3 = 700, }

    $x_1, x_2, x_3, x_4, s_1, s_2, s_3, A_1, A_2, A_3  \geq 0$
  \par}


  Phase I.

  \[ \min W = A_1 + A_2 + A_3 \]

    {\centering
      s.t.
    
      \sysdelim..\sysalign{r,r}%
      \systeme[x_1x_2x_3x_4s_1s_2s_3A_1A_2A_3]{%
        3x_1 + 4x_2 + 8x_3 + 6x_4 - s_1 + A_1 = 800,
        2x_1 + 2x_2 + 7x_3 + 5x_4 - s_2 + A_2 = 200,
        6x_1  +4x_2 + 7x_3 + 4x_4 - s_3 + A_3 = 700, }
    

    $x_1, x_2, x_3, x_4, s_1, s_2, s_3, A_1, A_2, A_3  \geq 0$
    \par}
  \end{solution}
